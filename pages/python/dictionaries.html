Dictionaries

Dictionaries are Python's built-in hash map — key-value pairs that allow O(1) average lookup, insertion, and deletion. Since Python 3.7, dicts preserve insertion order. This chapter covers all methods, dict comprehensions, nesting, and the powerful collections module extras.

1 · Creating Dictionaries

Python · Dict Creation 
# Using curly braces
person = {
    "name": "Alice",
    "age": 25,
    "city": "New York"
}

# Using dict() constructor
config = dict(host="localhost", port=8080, debug=True)

# Empty dict
d = {}
d2 = dict()

# From list of key-value pairs
pairs = [("a", 1), ("b", 2), ("c", 3)]
d3 = dict(pairs)

# From two lists using zip
keys   = ["x", "y", "z"]
values = [10, 20, 30]
d4 = dict(zip(keys, values))
print(d4)  # {'x': 10, 'y': 20, 'z': 30}

# fromkeys — all same value
template = dict.fromkeys(["a", "b", "c"], 0)
print(template)  # {'a': 0, 'b': 0, 'c': 0}
▶ Output
{'x': 10, 'y': 20, 'z': 30}
{'a': 0, 'b': 0, 'c': 0}

2 · Accessing & Modifying

Python · Access & Modify 
person = {"name": "Alice", "age": 25, "city": "NY"}

# Access by key
print(person["name"])       # "Alice"  — KeyError if not found!
print(person.get("age"))    # 25 — safe, returns None if missing
print(person.get("email", "N/A"))  # "N/A" — default value

# Modify / Add
person["age"] = 26           # update existing
person["email"] = "a@b.com" # add new key

# Delete
del person["city"]           # raises KeyError if not found
popped = person.pop("email") # removes & returns value
print(popped)                 # "a@b.com"

# Merge dicts (Python 3.9+)
d1 = {"a": 1, "b": 2}
d2 = {"b": 99, "c": 3}
merged = d1 | d2              # {'a':1, 'b':99, 'c':3} — d2 wins
d1 |= d2                      # update d1 in-place

# Check membership (only checks keys)
print("name" in person)      # True
print("Alice" in person)     # False (values not checked!)
▶ Output
Alice
25
N/A
a@b.com
True
False

3 · Dictionary Methods

Method Returns Description
d.get(k, default) value or default Safe access (no KeyError)
d.keys() dict_keys view All keys (live view)
d.values() dict_values view All values (live view)
d.items() dict_items view All (key, value) pairs
d.pop(k) value Remove & return value
d.popitem() (key, value) Remove & return last pair
d.update(d2) None Merge d2 into d
d.setdefault(k, v) value Get key; set to v if missing
d.copy() dict Shallow copy
d.clear() None Remove all items

4 · Iterating Dictionaries

Python · Dict Iteration 
scores = {"Alice": 89, "Bob": 76, "Carol": 93}

# Iterate keys (default)
for name in scores:
    print(name)            # Alice, Bob, Carol

# Iterate values
for score in scores.values():
    print(score)           # 89, 76, 93

# Iterate key-value pairs (MOST COMMON)
for name, score in scores.items():
    print(f"{name}: {score}")

# Sorted by key
for name in sorted(scores):
    print(name, scores[name])

# Sorted by value
for name, score in sorted(scores.items(), key=lambda x: x[1]):
    print(name, score)
▶ Output
Alice: 89
Bob: 76
Carol: 93

5 · Dictionary Comprehensions

Python · Dict Comprehensions 
# {key_expr: value_expr for item in iterable}

# Squares dict
squares = {x: x**2 for x in range(6)}
print(squares)  # {0:0, 1:1, 2:4, 3:9, 4:16, 5:25}

# Filter + transform
scores = {"Alice":89, "Bob":55, "Carol":93, "Dave":40}
passed = {name: score for name, score in scores.items() if score >= 60}
print(passed)   # {'Alice': 89, 'Carol': 93}

# Invert a dict (swap keys and values)
inverted = {v: k for k, v in scores.items()}
print(inverted) # {89:'Alice', 55:'Bob', 93:'Carol', 40:'Dave'}

# Word frequency counter
words = ["apple", "banana", "apple", "cherry", "banana", "apple"]
freq  = {word: words.count(word) for word in set(words)}
print(freq)     # {'apple':3, 'banana':2, 'cherry':1}
▶ Output
{0: 0, 1: 1, 2: 4, 3: 9, 4: 16, 5: 25}
{'Alice': 89, 'Carol': 93}
{'apple': 3, 'banana': 2, 'cherry': 1}

6 · collections.defaultdict & Counter

Python · collections module 
from collections import defaultdict, Counter, OrderedDict

# defaultdict — auto-creates missing keys with a default
word_groups = defaultdict(list)
words = [("a", "apple"), ("b", "banana"), ("a", "avocado")]
for letter, word in words:
    word_groups[letter].append(word)   # No KeyError!
print(dict(word_groups))
# {'a': ['apple', 'avocado'], 'b': ['banana']}

# Counter — count elements in any iterable
text = "hello world"
char_count = Counter(text)
print(char_count.most_common(3))   # [('l', 3), ('o', 2), ('h', 1)]

words = ["apple", "banana", "apple", "cherry", "apple"]
c = Counter(words)
print(c["apple"])       # 3
print(c.most_common(1)) # [('apple', 3)]

# Counter arithmetic
c1 = Counter({"a": 3, "b": 2})
c2 = Counter({"a": 1, "c": 4})
print(c1 + c2)           # Counter({'c':4,'a':4,'b':2})
▶ Output
{'a': ['apple', 'avocado'], 'b': ['banana']}
[('l', 3), ('o', 2), ('h', 1)]
3
[('apple', 3)]
Counter({'c': 4, 'a': 4, 'b': 2})
