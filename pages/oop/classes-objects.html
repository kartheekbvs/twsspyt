Classes & Objects

Object-Oriented Programming (OOP) organizes code around objects — bundles of data (attributes) and behavior (methods). A class is the blueprint; an object is an instance. Python is a fully object-oriented language — everything is an object, including integers, strings, and functions.

1 · Defining a Class

Python · Basic Class 
class BankAccount:
    """A simple bank account model."""

    bank_name = "PyBank"  # ← Class attribute (shared by all instances)

    def __init__(self, owner: str, balance: float = 0.0):
        # self refers to the instance being created
        self.owner   = owner    # ← Instance attribute
        self.balance = balance  # ← Instance attribute
        self._transactions = []  # ← "private" by convention

    def deposit(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Deposit must be positive")
        self.balance += amount
        self._transactions.append(("deposit", amount))

    def withdraw(self, amount: float) -> None:
        if amount > self.balance:
            raise ValueError("Insufficient funds")
        self.balance -= amount
        self._transactions.append(("withdrawal", amount))

    def __repr__(self) -> str:
        return f"BankAccount(owner='{self.owner}', balance={self.balance:.2f})"

# Create instances (objects)
alice = BankAccount("Alice", 1000.0)
bob   = BankAccount("Bob")             # balance defaults to 0.0

alice.deposit(500)
alice.withdraw(200)

print(alice)          # BankAccount(owner='Alice', balance=1300.00)
print(alice.balance)  # 1300.0
print(BankAccount.bank_name)  # PyBank  ← access class attribute
print(alice.bank_name)        # PyBank  ← also works on instance
▶ Output
BankAccount(owner='Alice', balance=1300.00)
1300.0

2 · Instance vs Class Attributes

Feature Instance Attribute Class Attribute
Defined in __init__ with self.x = ... Class body, outside methods
Unique per instance? ✅ Yes — each instance has its own ❌ No — shared by ALL instances
Access obj.attr ClassName.attr or obj.attr
Use case State that varies (name, balance) Constants, counters, defaults
Python · Class vs Instance Attributes 
class Dog:
    species = "Canis familiaris"  # class attribute
    count   = 0                   # tracks total dogs created

    def __init__(self, name, breed):
        self.name  = name    # instance attribute
        self.breed = breed   # instance attribute
        Dog.count += 1      # update class counter

fido  = Dog("Fido", "Labrador")
buddy = Dog("Buddy", "Poodle")

print(fido.species)   # "Canis familiaris" (from class)
print(fido.name)      # "Fido" (from instance)
print(Dog.count)      # 2 — shared class attribute
print(fido.count)     # 2 — also accessible via instance
▶ Output

Fido

3 · Three Types of Methods

Python · Instance, Class & Static Methods 
class Temperature:
    absolute_zero = -273.15   # in Celsius

    def __init__(self, celsius):
        self.celsius = celsius

    # ── Instance method — receives self ──────────────────
    def to_fahrenheit(self) -> float:
        return (self.celsius * 9/5) + 32

    # ── Class method — receives cls, not self ─────────────
    @classmethod
    def from_fahrenheit(cls, fahrenheit: float) -> "Temperature":
        """Alternative constructor."""
        celsius = (fahrenheit - 32) * 5/9
        return cls(celsius)     # creates a new instance

    # ── Static method — no self or cls ────────────────────
    @staticmethod
    def is_hot(celsius: float) -> bool:
        """No access to instance or class data."""
        return celsius > 30

# Usage
t = Temperature(100)
print(t.to_fahrenheit())       # 212.0

t2 = Temperature.from_fahrenheit(32)
print(t2.celsius)              # 0.0

print(Temperature.is_hot(35))  # True
print(Temperature.is_hot(20))  # False
▶ Output
212.0
0.0
True
False
Method Type First Arg Decorator Can access
Instance method self (none) Instance attrs + class attrs
Class method cls @classmethod Class attrs only (not instance)
Static method (none) @staticmethod Neither — like a plain function

4 · Properties (@property)

Properties allow controlled access to attributes — add validation or computation without changing the calling interface.

Python · @property Decorator 
class Circle:
    def __init__(self, radius: float):
        self._radius = radius   # "private" by convention

    @property
    def radius(self) -> float:
        """Getter — called when you ACCESS the property."""
        return self._radius

    @radius.setter
    def radius(self, value: float):
        """Setter — called when you ASSIGN the property."""
        if value < 0:
            raise ValueError("Radius cannot be negative!")
        self._radius = value

    @property
    def area(self) -> float:
        """Computed property — no setter needed."""
        import math
        return math.pi * self._radius ** 2

c = Circle(5)
print(c.radius)      # 5  (calls getter)
print(c.area)        # 78.54  (computed)

c.radius = 10        # calls setter — validated!
print(c.radius)      # 10

c.radius = -3        # raises ValueError!
▶ Output
5
78.53981633974483
10
ValueError: Radius cannot be negative!

5 · dataclasses (Python 3.7+)

@dataclass automatically generates __init__, __repr__, and __eq__ for you:

Python · @dataclass 
from dataclasses import dataclass, field
from typing import List

@dataclass
class Student:
    name: str
    grade: int
    scores: List[float] = field(default_factory=list)
    gpa: float = 0.0

    def average_score(self) -> float:
        return sum(self.scores) / len(self.scores) if self.scores else 0.0

# __init__ auto-generated: Student(name, grade, scores=[], gpa=0.0)
s = Student("Alice", 10, scores=[89, 92, 85], gpa=3.8)
print(s)     # Student(name='Alice', grade=10, scores=[89,92,85], gpa=3.8)
print(s.average_score())   # 88.67

# Equality works out of the box!
s2 = Student("Alice", 10, scores=[89, 92, 85], gpa=3.8)
print(s == s2)   # True
▶ Output

True
